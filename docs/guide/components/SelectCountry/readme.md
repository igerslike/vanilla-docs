## Example & Snippet
```vue
<template>
    <VanillaSelectCountry
        v-model="value"
        placeholder="Select your country"
    />
</template>
<script lang="ts">
import { VanillaSelectCountry } from '@vanilla-components';
import { defineComponent, ref } from 'vue';
export default defineComponent({
    components: {
		    VanillaSelectCountry
    },
    setup() {
        const value = ref(null);
        return {
            value,
        };
    },
});
</script>
```
